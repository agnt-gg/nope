Quick Start 
import { NOPE } from '@agnt-gg/nope';

const nope = NOPE.preset('standard');

// Scan context + MCP tools
nope.scan(contextText);
nope.scan({ name: 'shell', description: 'Run commands' });

// Wrap tools — everything guarded
const safe = nope.wrap(plug.tools);
